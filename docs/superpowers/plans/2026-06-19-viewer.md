# QRTrans Viewer（全屏播放器）实现计划

> **面向 AI 代理的工作者：** 必需子技能：使用 superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans 逐任务实现此计划。步骤使用复选框（`- [ ]`）跟踪进度。

**目标：** 独立的全屏图片序列播放器 `qrtrans-viewer`，针对 colormatrix/QR 帧的"显示→对端拍摄/截图"优化：真全屏 1:1、键盘切换、按间隔自动切换；并在启动时把实测分辨率写入 `./qrtrans.json` 供 encoder 读取（密度最大化）。通用图片播放器，不耦合 qrtrans 内部。

**架构：** Tkinter + Pillow（stdlib，Win7 兼容）。`qrtrans_viewer/core.py`（纯逻辑：列表/状态机/间隔/config 读写，全单测）+ `qrtrans_viewer/gui.py`（Tkinter 接线，薄）+ `__main__.py`。独立 PyInstaller spec（windowed）。encoder 侧小改：未显式传 `--screen` 时读 `./qrtrans.json`。

**技术栈：** Tkinter、Pillow、PyInstaller（GHA windows-latest / Python 3.8）。

**参考规格：** `docs/superpowers/specs/2026-06-19-viewer-design.md`

**验证：** `.venv/bin/pytest -q`（core 全测）；GUI 冒烟（需图形环境，Linux 无头跳过/Windows CI 跑）；frozen viewer exe 冒烟。

CONFIG_FILENAME = "qrtrans.json"   # CWD 本地
def list_images(path: Path) -> List[Path]          # 目录/单图 → 排序后的图片列表

```python
from __future__ import annotations
import json
from dataclasses import dataclass, field
from datetime import datetime, timezone
from pathlib import Path
from typing import List, Optional, Tuple

IMAGE_SUFFIXES = {".png", ".jpg", ".jpeg", ".bmp", ".gif"}
CONFIG_FILENAME = "qrtrans.json"
MIN_INTERVAL = 0.2


def list_images(path: Path) -> List[Path]:
    if path.is_file():
        return [path]
    if not path.is_dir():
        return []
    return sorted(p for p in path.iterdir() if p.suffix.lower() in IMAGE_SUFFIXES)


@dataclass
class ViewerState:
    images: List[Path]
    index: int = 0
    playing: bool = False
    interval: float = 3.0
    loop: bool = False

    def _cur(self) -> Optional[Path]:
        return self.images[self.index] if self.images else None

    def first(self) -> Optional[Path]:
        self.index = 0
        return self._cur()

    def last(self) -> Optional[Path]:
        self.index = max(0, len(self.images) - 1)
        return self._cur()

    def next(self) -> Optional[Path]:
        if not self.images:
            return None
        if self.index < len(self.images) - 1:
            self.index += 1
        elif self.loop:
            self.index = 0
        return self._cur()

    def prev(self) -> Optional[Path]:
        if not self.images:
            return None
        if self.index > 0:
            self.index -= 1
        elif self.loop:
            self.index = len(self.images) - 1
        return self._cur()

    def bump_interval(self, delta: float) -> float:
        self.interval = max(MIN_INTERVAL, self.interval + delta)
        return self.interval

    def advance(self) -> Optional[Path]:
        """自动切换一步：仅当 playing。末尾非 loop → 停止并暂停；loop → 回首页。"""
        if not self.playing or not self.images:
            return None
        if self.index < len(self.images) - 1:
            self.index += 1
            return self._cur()
        if self.loop:
            self.index = 0
            return self._cur()
        self.playing = False
        return None


def write_config(screen: Tuple[int, int], path: str = CONFIG_FILENAME) -> None:
    """best-effort：写失败静默忽略。"""
    data = {"screen": [int(screen[0]), int(screen[1])],
            "recorded_at": datetime.now(timezone.utc).isoformat()}
    try:
        Path(path).write_text(json.dumps(data), encoding="utf-8")
    except OSError:
        pass


def read_config(path: str = CONFIG_FILENAME) -> Optional[Tuple[int, int]]:
    try:
        raw = Path(path).read_text(encoding="utf-8")
        data = json.loads(raw)
        w, h = int(data["screen"][0]), int(data["screen"][1])
        return (w, h)
    except (OSError, ValueError, KeyError, TypeError, IndexError):
        return None
```

```python
from __future__ import annotations
import sys
from pathlib import Path
from tkinter import Tk, Label
from PIL import Image, ImageTk
from .core import ViewerState, list_images, write_config


def run(target: Path, interval: float, loop: bool, overlay: bool) -> int:
    images = list_images(target)
    if not images:
        print("error: no images found", file=sys.stderr)
        return 2
    root = Tk()
    # 实测分辨率 + 写 config（best-effort）
    root.update_idletasks()
    screen = (root.winfo_screenwidth(), root.winfo_screenheight())
    write_config(screen)

    state = ViewerState(images=images, index=0, playing=False, interval=interval, loop=loop)

    root.attributes("-fullscreen", True)
    try:
        root.overrideredirect(True)
        root.geometry(f"{screen[0]}x{screen[1]}+0+0")
    except Exception:
        pass
    root.configure(bg="black")

    img_label = Label(root, bg="black")
    img_label.pack(fill="both", expand=True)
    info_label = Label(root, bg="black", fg="white", anchor="sw", justify="left")
    info_label.place(x=8, rely=1.0, y=-8, anchor="sw")

    _photo = {"obj": None}      # 防 GC
    _next_photo = {"obj": None}

    def _render():
        cur = images[state.index]
        im = Image.open(cur)
        im.load()
        _photo["obj"] = ImageTk.PhotoImage(im)
        img_label.configure(image=_photo["obj"])
        # 预加载下一张
        if state.index + 1 < len(images):
            try:
                _next_photo["obj"] = ImageTk.PhotoImage(Image.open(images[state.index + 1]))
            except Exception:
                _next_photo["obj"] = None
        if overlay:
            mark = "▶" if state.playing else "⏸"
            info_label.configure(text=f"{cur.name} · {state.index+1}/{len(images)} · "
                                      f"{state.interval:.1f}s · {mark}")
            info_label.lift()
        else:
            info_label.configure(text="")

    _timer = {"id": None}

    def _schedule():
        if _timer["id"] is not None:
            root.after_cancel(_timer["id"])
        if state.playing:
            _timer["id"] = root.after(int(state.interval * 1000), _tick)

    def _tick():
        if state.advance() is not None:
            _render()
            _schedule()
        else:
            _render()   # 末尾已自动暂停

    def _go_next():
        state.next(); _render(); _schedule()

    def _go_prev():
        state.prev(); _render(); _schedule()

    def _toggle_play():
        state.playing = not state.playing
        _render(); _schedule()

    def _quit():
        root.destroy()

    root.bind("<Right>", lambda e: _go_next())
    root.bind("<space>", lambda e: _go_next())
    root.bind("<Left>", lambda e: _go_prev())
    root.bind("p", lambda e: _toggle_play())
    root.bind("P", lambda e: _toggle_play())
    root.bind("<plus>", lambda e: (state.bump_interval(0.5), _render(), _schedule()))
    root.bind("<equal>", lambda e: (state.bump_interval(0.5), _render(), _schedule()))
    root.bind("<minus>", lambda e: (state.bump_interval(-0.5), _render(), _schedule()))
    root.bind("o", lambda e: None)   # overlay 切换见下（简化：重构为可变状态）
    root.bind("<Home>", lambda e: (state.first(), _render(), _schedule()))
    root.bind("<End>", lambda e: (state.last(), _render(), _schedule()))
    root.bind("<Escape>", lambda e: _quit())
    root.bind("q", lambda e: _quit())
    root.bind("Q", lambda e: _quit())

    _render()
    root.mainloop()
    return 0
```

> overlay 开关（`O` 键）：上面 `o` 绑定留了占位；实现时用一个可变 `overlay_on` 变量在绑定里翻转并 `_render()`。实现者补全。

import argparse
import sys
from pathlib import Path
from .gui import run


def main(argv=None):
    p = argparse.ArgumentParser(prog="qrtrans-viewer",
                                description="全屏图片序列播放器（为 colormatrix/QR 帧优化）")
    p.add_argument("target", type=Path, help="目录或单图")
    p.add_argument("--interval", type=float, default=3.0, help="自动切换间隔秒（默认 3.0）")

    p.add_argument("--no-overlay", dest="overlay", action="store_false", default=True)
    args = p.parse_args(argv)
    if not args.target.exists():
        print(f"error: not found: {args.target}", file=sys.stderr)
        return 2
    return run(args.target, args.interval, args.loop, args.overlay)


if __name__ == "__main__":
    sys.exit(main())
